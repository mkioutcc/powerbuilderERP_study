常用 PowerBuilder 程式碼範例

// 設定資料庫連線參數 (以 SQL Server 為例)
SQLCA.DBMS = "SNC SQL Server"
SQLCA.Database = "ERP_Database"
SQLCA.ServerName = "ServerName"
SQLCA.LogId = "username"
SQLCA.LogPass = "password"
SQLCA.AutoCommit = False

// 建立資料庫連線
CONNECT USING SQLCA;

IF SQLCA.SQLCode < 0 THEN
    MessageBox("連線錯誤", "無法連接到資料庫: " + SQLCA.SQLErrText)
    RETURN
END IF

// 執行查詢
String ls_sql = "SELECT customer_id, customer_name FROM customers"
DECLARE cust_cursor CURSOR FOR SQLSA;
PREPARE SQLSA FROM :ls_sql;
OPEN cust_cursor;

// 處理查詢結果
FETCH cust_cursor INTO :ll_id, :ls_name;
DO WHILE SQLCA.SQLCode = 0
    // 處理資料
    FETCH cust_cursor INTO :ll_id, :ls_name;
LOOP

// 關閉資料庫連線
CLOSE cust_cursor;
DISCONNECT USING SQLCA;
// 綁定 DataWindow 控制項到資料來源
dw_customer.SetTransObject(SQLCA)

// 檢索資料
Long ll_rows
ll_rows = dw_customer.Retrieve()

IF ll_rows = 0 THEN
    MessageBox("提示", "沒有符合條件的資料")
ELSEIF ll_rows < 0 THEN
    MessageBox("錯誤", "檢索資料時發生錯誤: " + String(ll_rows))
END IF

// 新增一筆資料
Long ll_row
ll_row = dw_customer.InsertRow(0)
dw_customer.SetItem(ll_row, "customer_name", "新客戶")
dw_customer.SetItem(ll_row, "contact_person", "聯絡人")
dw_customer.SetItem(ll_row, "phone", "02-12345678")

// 更新資料庫
IF dw_customer.Update() = 1 THEN
    COMMIT USING SQLCA;
    MessageBox("成功", "資料已成功更新")
ELSE
    ROLLBACK USING SQLCA;
    MessageBox("錯誤", "更新資料失敗: " + SQLCA.SQLErrText)
END IF
// 創建 HTTP 客戶端物件
OLEObject lole_http
lole_http = CREATE OLEObject

// 連接到 MSXML2.XMLHTTP 物件
Integer li_return
li_return = lole_http.ConnectToObject("MSXML2.XMLHTTP")

IF li_return < 0 THEN
    MessageBox("錯誤", "無法創建 HTTP 物件")
    RETURN
END IF

// 設定 HTTP 請求
String ls_url = "https://api.example.com/orders"
String ls_method = "GET"
lole_http.Open(ls_method, ls_url, False)
lole_http.SetRequestHeader("Content-Type", "application/json")

// 發送請求
lole_http.Send()

// 處理回應
Integer li_status = lole_http.Status
String ls_response = lole_http.ResponseText

IF li_status = 200 THEN
    // 解析 JSON 回應 (需要 JSON 解析器)
    // ...處理回應資料
ELSE
    MessageBox("API 錯誤", "HTTP 狀態碼: " + String(li_status))
END IF

// 釋放資源
DESTROY lole_http
